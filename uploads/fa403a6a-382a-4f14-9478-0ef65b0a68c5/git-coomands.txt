git init - Начало работы гита.
git status - Проверка статуса файла.
git add ./название-файла.расширение
git commit -m "Текст commita"
git push - пушит проект на уд.репо
git pull - стягивает с уд.репозитория
git branch - пока ветки вашего проекта
git checkout - переключает на ветку
git merge - слияние веток
git reset (--hard) - откат до коммита
git branch (-d) - удаление ветки
git checkout -b iss53 - создает и сразу переключается 
git log (-2) - просмотр истории коммитов
